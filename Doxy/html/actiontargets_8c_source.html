398 private int FindIndexForStoring(float value)
399 {
400 int left = 0;
401 int right = m_Targets.Count() - 1;
402 while ( left <= right )
403 {
404 int middle = (left + right) / 2;
405 float middleValue = m_Targets.Get(middle).GetUtility();
406
407 if ( middleValue == value )
408 return middle;
409 else if ( middleValue < value )
410 right = middle - 1;
411 else
412 left = middle + 1;
413 }
414
415 return left;
416 }

419 private float ComputeUtility(Object pTarget, vector pRayStart, vector pRayEnd, Object cursorTarget, vector hitPos)
420 {
422 if (vector.DistanceSq(hitPos, m_Player.GetPosition()) > c_MaxTargetDistance * c_MaxTargetDistance)
423 return -1;
424
425 if (pTarget)
426 {
427 if ( pTarget == cursorTarget )
428 {
430 if ( pTarget.GetType() == string.Empty )
431 return 0.01;
432
433 if ( pTarget.IsBuilding() )
434 return 0.25;
435
436 if ( pTarget.IsTransport() )
437 return 0.25;
438
439 if ( pTarget.IsWell() )
440 return 0.9;
441
442 vector playerPosXZ = m_Player.GetPosition();
443 vector hitPosXZ = hitPos;
444 playerPosXZ[1] = 0;
445 hitPosXZ[1] = 0;
446 if ( vector.DistanceSq(playerPosXZ, hitPosXZ) <= c_MaxTargetDistance * c_MaxTargetDistance )
447 return c_UtilityMaxValue;
448 }
449
450 float distSqr = DistSqrPoint2Line(pTarget.GetPosition(), pRayStart, pRayEnd);
451 return (c_UtilityMaxDistFromRaySqr - distSqr) / c_UtilityMaxDistFromRaySqr;
452 }
453
454 return -1;
455 }

458 private float DistSqrPoint2Line(vector pPoint, vector pL1, vector pL2)
459 {
460 vector v = pL2 - pL1;
461 vector w = pPoint - pL1;
462
463 float c1 = vector.Dot(w,v);
464 float c2 = vector.Dot(v,v);
465
466 if ( c1 <= 0 || c2 == 0 )
467 return vector.DistanceSq(pPoint, pL1);
468
469 float b = c1 / c2;
470 vector nearestPoint = pL1 + (v * b);
471 return vector.DistanceSq(pPoint, nearestPoint);
472 }

506 {
507 int numObstructed = 0;
508 int mCount = m_VicinityObjects.Count();
509 IsObjectObstructedCache cache = new IsObjectObstructedCache(m_RayStart, mCount);
510 mCount--;
511
513 for ( int i = mCount; i >= 0; --i )
514 {
515 Object object = m_VicinityObjects.GetObject(i);
516 Object parent = m_VicinityObjects.GetParent(i);
517
519 if (object && !parent)
520 {
523 if (numObstructed > OBSTRUCTED_COUNT_THRESHOLD && object != cursor_target)
524 {
525 m_VicinityObjects.Remove(object);
526 continue;
527 }
528
530 if (object != cursor_target && IsObstructedEx(object, cache))
531 {
532 m_VicinityObjects.Remove(object);
533 numObstructed++;
534 }
535
536 cache.ClearCache();
