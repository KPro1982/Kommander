84 protected void ProcessUniversalTemperatureSources()
85 {

88 m_UTSAverageTemperature = 0;

93 array<float> utsTemperatures = new array<float>();
94
95 // get temperature from the source (based on distance), save it for min/max filtering
96 for (int i = 0; i < m_UTemperatureSourceDebugs.Count(); i++)
97 {
98 UniversalTemperatureSourceDebug utsd = m_UTemperatureSourceDebugs[i];
100 if (vector.DistanceSq(m_Player.GetPosition(), utsd.GetValue(0).ToVector()) > Math.SqrFloat(utsd.GetValue(2).ToFloat()))
101 {
102 continue;
103 }
104
105 utsTemperatures.Insert(CalcTemperatureFromTemperatureSource(m_UTemperatureSourceDebugs[i]));
106 }
107
108 float min = MiscGameplayFunctions.GetMinValue(utsTemperatures);
109 float max = MiscGameplayFunctions.GetMaxValue(utsTemperatures);
110
111 if (max > 0 && min < 0)
112 {
113 m_UTSAverageTemperature = (max + min) * 0.5;
114 }
115 else
116 {
117 m_UTSAverageTemperature = max;

122 protected float CalcTemperatureFromTemperatureSource(notnull UTemperatureSourceDebug utsd)
123 {
124 float distance = vector.Distance(m_Player.GetPosition(), utsd.GetValue(0).ToVector());
125 distance = Math.Max(distance, 0.1); //min distance cannot be 0 (division by zero)
126 float temperature = 0;
127
129 if (distance > utsd.GetValue(1).ToFloat())
130 {
131 float distFactor = 1 - (distance / utsd.GetValue(2).ToFloat());
132 distFactor = Math.Max(distFactor, 0.0);
133 temperature = utsd.GetValue(3).ToFloat() * distFactor;
134 }
135 else
136 {
137 temperature = utsd.GetValue(3).ToFloat();
138 }
139
140 //Print(temperature);
141
142 return temperature;
