312 static int RegisterParticle(string root_path, string file_name)
313 {
314 if (!m_ParticlePaths)
315 m_ParticlePaths = new map<int, string>;
316
317 if (!m_ParticleNames)
318 m_ParticleNames = new map<string, int>;
319
320 string path = string.Format("%1%2", root_path, file_name);
321 string fullPath = string.Format("%1.ptc", path);
322
323 if ( !FileExist(fullPath) )
324 {
325 string devMsgSuffix;
326 #ifdef DEVELOPER
327 devMsgSuffix = " Then rebuild 'scripts' and 'graphics' PBOs.";
328 #endif
329 ErrorEx(string.Format("Particle file not found! <%1> Correct path to this particle in script file ParticleList.c!%2",
330 fullPath, devMsgSuffix), ErrorExSeverity.WARNING);
331 }
332
333 int existingNameId;
334 if (m_ParticleNames.Find(file_name, existingNameId))
335 {
336 int existingPathId;
337 if (m_ParticlePaths.GetKeyByValueChecked(path, existingPathId))
338 {
339 if (existingNameId == existingPathId)
340 {
341 return existingNameId; // We have already registered this exact file
342 }
343 else
344 {
345 ErrorEx(string.Format("ParticleNames and ParticlePaths array are containing conflicting data. ID %1 vs ID %2", existingNameId, existingPathId));
346 }
347 }
348 else
349 {
350 ErrorEx(string.Format("Duplicate particle filename: %1; GetParticleIDByName will only work with the first path if there are two different paths.", file_name), ErrorExSeverity.WARNING);
351 }
352 }
353 else
354 {
355 m_ParticleNames.Insert(file_name, m_lastID + 1);
356 }
357
358 ++m_lastID;
359 m_ParticlePaths.Insert(m_lastID, path);
360 return m_lastID;
361 }

390 static int GetParticleID(string particle_file)
391 {
