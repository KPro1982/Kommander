105 {

404 static void GetFiltredConfigClasses( string search_string, out TStringArray filtered_classes, bool only_public = true )
405 {
406 TStringArray searching_in = new TStringArray;
407 GetBaseConfigClasses( searching_in );
408
409 filtered_classes.Clear();
410
411 search_string.ToLower();
412
413 for ( int s = 0; s < searching_in.Count(); ++s )
414 {
415 string config_path = searching_in.Get(s);
416
417 int objects_count = GetGame().ConfigGetChildrenCount(config_path);
418 for (int i = 0; i < objects_count; i++)
419 {
420 string childName;
421 GetGame().ConfigGetChildName(config_path, i, childName);
422
423 if ( only_public )
424 {
425 int scope = GetGame().ConfigGetInt( config_path + " " + childName + " scope" );
426 if ( scope == 0 )
427 {
428 continue;
429 }
430 }
431
432 string nchName = childName;
433 nchName.ToLower();
434
435 if ( nchName.Contains(search_string) != -1)
436 {
437 filtered_classes.Insert(childName);
438 }
439 }
440 }

448 static private void LogMessage(string level, string plugin, string entity, string author, string label, string message)
449 {
450 if (GetGame() == NULL || LogManager.IsLogsEnable() == false )
451 {
452 return;
453 }
454
455 bool is_server_log = ( GetGame().IsServer() && GetGame().IsMultiplayer() );
456
457
458 // Formation output to external file
459 // %date{MM-dd HH:mm:ss} | %Enviroment | %Level | %Module | %Entity | %Author | %Label | %Message
460 string date = GetDate();
461 string env = "Client";
462 string msg = string.Empty;
463
464 if ( is_server_log )
465 {
466 env = "Server";
467 }
468
469 msg = string.Format("%1 | %2 | %3 | %4 | %5 | %6 | %7", date, env, level, plugin, entity, label, message);
470
471 if ( is_server_log )
472 {
473 SaveLog(msg);
474#ifdef DEVELOPER //not sendig log to clients on stable
475 Param1<string> msg_p = new Param1<string>(msg);
476 CallMethod(CALL_ID_SEND_LOG, msg_p);
477#endif
478 }
479 else
480 {
481 SaveLog(msg);
482 }
483 }
