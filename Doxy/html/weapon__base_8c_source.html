682 bool FillInnerMagazine( string ammoType = "", int flags = WeaponWithAmmoFlags.CHAMBER )
683 {
684 // Don't try to fill it when there are none
685 if (!HasInternalMagazine(-1))
686 return false;
687
688 // Make sure the given ammoType is actually useable
689 if (ammoType != "")
690 {
691 if (!AmmoTypesAPI.MagazineTypeToAmmoType(ammoType, ammoType))
692 return false;
693 }
694
695
696 bool didSomething = false;
697 int muzzCount = GetMuzzleCount();
698
699 bool ammoRng = ammoType == "";
700 bool ammoFullRng = ammoRng && (flags & WeaponWithAmmoFlags.AMMO_MAG_RNG);
701
702 // No full RNG flag, so pick one random and use only this one
703 if (ammoRng && !ammoFullRng)
704 ammoType = GetRandomChamberableAmmoTypeName(0);
705
706 // Fill the internal magazine
707 for (int i = 0; i < muzzCount; ++i)
708 {
709 int ammoCount = GetInternalMagazineMaxCartridgeCount(i);
710
711 // Decide random quantity when enabled
712 if ( flags & WeaponWithAmmoFlags.QUANTITY_RNG )
713 ammoCount = Math.RandomIntInclusive(0, ammoCount);
714
715 // Only do the things when there is actually ammo to fill
716 if (ammoCount > 0)
717 {
718 // Push in the cartridges
719 for (int j = 0; j < ammoCount; ++j)
720 {
721 // Full random, decide a new one for every cartridge
722 if ( ammoFullRng )
723 ammoType = GetRandomChamberableAmmoTypeName(i);
724
725 PushCartridgeToInternalMagazine(i, 0, ammoType);
726 didSomething = true;
727 }
728 }
729 }
730
731 // Call the chamber method if asked for
732 bool chamber = (flags & WeaponWithAmmoFlags.CHAMBER) || (flags & WeaponWithAmmoFlags.CHAMBER_RNG);
733 if (chamber && FillChamber(ammoType, flags))
734 {
735 didSomething = true;
736 }
737
738 // Does not need any FSM fixing, FSM does not care about inner magazines
739
740 return didSomething;

750 bool FillChamber( string ammoType = "", int flags = WeaponWithAmmoFlags.CHAMBER )
751 {
752 // Quickly check if there are any chambers we can fill
753 int muzzCount = GetMuzzleCount();
754 bool anyEmpty = false;
755
756 for (int m = 0; m < muzzCount; ++m)
757 {
758 if (IsChamberEmpty(m))
759 {
760 anyEmpty = true;
761 break;
762 }
763 }
764
765 if (!anyEmpty)
766 return false;
767
768 // Make sure the given ammoType is actually useable
769 if (ammoType != "")
770 if (!AmmoTypesAPI.MagazineTypeToAmmoType(ammoType, ammoType))
771 return false;
772
773 // Just so we don't '&' wastefully in a loop
774 bool didSomething = false;
775 bool chamberFullRng = (flags & WeaponWithAmmoFlags.CHAMBER_RNG_SPORADIC);
776 bool chamberRng = (flags & WeaponWithAmmoFlags.CHAMBER_RNG);
777 bool chamber = (flags & WeaponWithAmmoFlags.CHAMBER);
778
779 if (chamber || chamberRng || chamberFullRng)
780 {
781 int amountToChamber = muzzCount;
782
783 // No need to do this for full rng, as that will roll for every muzzle
784 if (chamberRng)
785 amountToChamber = Math.RandomIntInclusive(0, muzzCount);
786
787 bool chamberAmmoRng = (ammoType == "");
788 bool chamberAmmoFullRng = chamberAmmoRng && (flags & WeaponWithAmmoFlags.AMMO_CHAMBER_RNG);
789
790 // No full RNG flag, so pick one random and use only this one
791 if (chamberAmmoRng && !chamberAmmoFullRng)
792 ammoType = GetRandomChamberableAmmoTypeName(0);
793
794 for (int i = 0; i < muzzCount; ++i)
795 {
796 // Skip when there's already something in the chamber
797 if (!IsChamberEmpty(i))
798 continue;
799
800 // Roll the rng when enabled
801 if (chamberFullRng)
802 chamber = Math.RandomIntInclusive(0, 1);
803
804 // We chambering
805 if (chamber)
806 {
807 // Full random, decide a new one for every muzzle
808 if ( chamberAmmoFullRng )
809 ammoType = GetRandomChamberableAmmoTypeName(i);
810
811 // Push it
812 PushCartridgeToChamber(i, 0, ammoType);
813 didSomething = true;
814
815 // Stop chambering when we hit the desired amount
816 --amountToChamber;
817 if (amountToChamber <= 0)
818 break;
819 }
820 }
821 }
822
823 // Only fix the FSM and Synchronize when absolutely needed
824 if (!didSomething)
825 return false;
826
827 // FSM cares about chamber state
828 RandomizeFSMState();
829 Synchronize();
830
831 return true;
832 }
