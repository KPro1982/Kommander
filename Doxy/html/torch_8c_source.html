285 void Upgrade(ItemBase source)
286 {
287 if (!GetRag())
288 {
289 return;
290 }
291 RuinRags();
292 LockRags(true);
293 float torchCurrentEnergy = GetCompEM().GetEnergy();
294 float sourceQuantity = 100000;//for upgrade from environment(gas pump)
295
296 if (source)
297 {
298 sourceQuantity = source.GetQuantity();
299 }
300
301 float maxOverallCapacity = GetRagQuantity() * 2 * m_BurnTimePerRagEx;
302 //float maxUpgradeCapacity = GetRagQuantity() * m_BurnTimePerRagEx;
303 float currentlyUpgraded = Math.Max(0, torchCurrentEnergy - (GetRagQuantity() * m_BurnTimePerRagEx));
304 //float freeUpgradeCapacity = maxUpgradeCapacity - currentlyUpgraded;
305 float freeUpgradeCapacity = maxOverallCapacity - torchCurrentEnergy;
306 float upgradeQuantity = Math.Min(freeUpgradeCapacity, sourceQuantity);
307 int upgradedRags = Math.Ceil((upgradeQuantity + currentlyUpgraded) / m_BurnTimePerRagEx);
308 upgradedRags = Math.Min(GetRagQuantity(), upgradedRags);
309 m_RagsUpgradedCount = upgradedRags;
310 GetCompEM().AddEnergy(upgradeQuantity);
311 m_ConsumeRagFlipFlop = 0;//consume fuel first
312 if (source)

314 source.AddQuantity(-upgradeQuantity);
315 }

317 UpdateCheckForReceivingUpgrade();
