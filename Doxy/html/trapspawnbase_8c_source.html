326 // Used to check catch conditions and spawn relevant prey
327 void SpawnCatch()
328 {
329 // Only server side, let's make sure
330 if ( GetGame().IsClient() )
331 return;
332
333 // If we can't get CEApi we won't be able to test proximity
334 if ( !GetCEApi() )
335 return;
336
337 int i;
338 m_Bait = null;
339
340 // We check if we have bait and can catch
341 m_CanCatch = SetCanCatch( m_Bait );
342
343 SetIsDeploySound( false );
344 SetSynchDirty();
345
346 // We check the distance from players through CEApi if that check is enabled
347 if ( m_MinimalDistanceFromPlayersToCatch > 0 )
348 {
349 if ( !GetCEApi().AvoidPlayer( GetPosition(), m_MinimalDistanceFromPlayersToCatch ) && m_CanCatch )
350 {
351 m_CanCatch = false;
352
353 // We could not catch yet, let's retest until we can
354 if ( m_Timer )
355 m_Timer.Stop();
356
357 m_Timer = new Timer( CALL_CATEGORY_GAMEPLAY );
358 m_Timer.Run( m_UpdateWaitTime, this, "SpawnCatch" );
359
360 return; // No need to go further, let's just wait for next check
361 }
362 }
363
364 // We get the probability to catch depending on bait presence
365 if ( m_Bait )
366 m_FinalCatchProb = m_BaitCatchProb;
367 else
368 m_FinalCatchProb = m_NoBaitCatchProb;
369
370 // We get the position of prey when it will spawn
371 m_PreyPos = GetPosition();
372 if ( MemoryPointExists("Prey_Position") )
373 {
374 m_PreyPos = ModelToWorld( GetMemoryPointPos("Prey_Position") );
375 }

379 // Also outs the current bait to check for specific catches if required
380 bool SetCanCatch( out EntityAI bait )
381 {
382 int slotIdx = InventorySlots.GetSlotIdFromString("Trap_Bait");
383 bait = GetInventory().FindAttachment( slotIdx );
384
385 if ( bait )
386 {
387 Edible_Base edibleBait = Edible_Base.Cast( bait );
388 if ( edibleBait )
389 {
390 if ( !edibleBait.GetFoodStage().IsFoodBurned() && !edibleBait.GetFoodStage().IsFoodRotten() )
391 return true;
392 else
393 return false; // We have invalid bait, no catch
394 }

439 bool IsSurfaceWater( vector position )
440 {
441 string surface_type;
