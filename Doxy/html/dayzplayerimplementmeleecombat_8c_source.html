334 protected void TargetSelection()
335 {
336 // Prepare common variables
337 PlayerBase player = PlayerBase.Cast(m_DZPlayer);
338 vector pos = m_DZPlayer.GetPosition();
339 vector rayStart = m_DZPlayer.GetBonePositionWS(m_DZPlayer.GetBoneIndexByName("Head"));
340
341 vector cameraPos, cameraRot, cameraDir;
342 m_DZPlayer.GetCurrentCameraTransform(cameraPos, cameraDir, cameraRot);
343 vector dir = MiscGameplayFunctions.GetHeadingVector(player);
344 dir[1] = cameraDir[1];
345
346 // Calculate max distances
347 float dist = GetRange();
348 float angle = GetAngle();
349 float dist2 = Math.SqrFloat(dist);
350
351 // There's generally 2 TargetSelection calls per animation
352 // First to obtain target to move towards during animation
353 // Second one is to see if that target is still in range before applying damage to it
354 // m_WasHit means the hit event occured, so this is the second call
355 if ( m_WasHit )
356 {
357 // See if the component is still in range
358 if (CanObjectBeTargeted(m_TargetObject) && ( vector.DistanceSq(rayStart, m_TargetObject.GetDamageZonePos(m_HitZoneName)) <= dist2 ))
359 {
360 m_AllTargetObjects.Insert(m_TargetObject);
361 // This result should still be cached, so no need to fill it in again
362 return;
363 }
364 }
365
366 // Find a new target
367 InternalResetTarget();
368
369 Object target;
370 vector hitPos;
371 int hitZone;
372
373 // First pass - Prefer what the player is looking at (only aligneable ones, we will try for nonaligneable later)
374 if (HitZoneSelectionRaycastHelper(hitPos, hitZone, target))
375 {
376 if (m_ForceUntargetable)
377 {
378 SetTarget(target, hitPos, hitZone);
379 return;
380 }
381
382 if (CanObjectBeTargeted(target) && vector.DistanceSq(rayStart, hitPos) <= dist2)
383 {
384 m_AllTargetObjects.Insert(target);
385
386 SetTarget(target, hitPos, hitZone);
387 return;
388 }
389 }
390
391 // Second pass - Try to obtain something aligneable closeby the player watching direction
392 MeleeTargetData targetData = m_MeleeTargeting.GetMeleeTargetEx(new MeleeTargetSettings(pos, dist*0.75, angle, TARGETING_MIN_HEIGHT, TARGETING_MAX_HEIGHT, rayStart, dir, TARGETING_RAY_RADIUS_EX, m_DZPlayer, m_TargetableObjects), m_AllTargetObjects, m_BlacklistedDamageZones);
393
394 if (targetData)
395 {
396 SetTarget(targetData.Obj, targetData.HitPos, targetData.HitComponent);
397 return;
398 }
399
400 // Third pass - We found no aligneable target, check if the target that we are directly looking at is nonalignable (big)
401 if (CanObjectBeTargeted(target, true) && vector.DistanceSq(rayStart, hitPos) <= dist2)
402 {
403 m_AllTargetObjects.Insert(target);
404
405 SetTarget(target, hitPos, hitZone);
406 return;
407 }

416 protected int TrySelectFinisherType(InventoryItem weapon, EntityAI target)
417 {
418 if (m_WasHit)
419 return -1;
420
422 if (target && target.CanBeBackstabbed() && weapon && (weapon.IsMeleeFinisher() || m_HitType == EMeleeHitType.WPN_STAB) && !weapon.IsRuined() )
423 {
424 bool playGenericFinisherAnimation = false;
425 ZombieBase targetZombie = ZombieBase.Cast(target);
426 if (targetZombie && m_DebugForcedFinisherType == -1)
427 {
429 if (!IsEntityBehindEntityInAngle(m_DZPlayer, target, 60))
430 {
431 return -1;
432 }
433
434 int mindState = targetZombie.GetMindStateSynced();
436 if (mindState >= DayZInfectedConstants.MINDSTATE_DISTURBED)
437 {
438 return -1;
439 }
440 }
441
442 PlayerBase targetPlayer = PlayerBase.Cast(target);
444 if (targetZombie)
445 {
446 playGenericFinisherAnimation = targetZombie.IsCrawling();
447 }
448 else if (targetPlayer)
449 {
450 playGenericFinisherAnimation = targetPlayer.IsInProne();
451 }
452 else
453 {
454 playGenericFinisherAnimation = true;
455 }
456
458 if (weapon.IsWeapon())
459 {
460 return EMeleeHitType.WPN_STAB_FINISHER;
461 }
462 else if (playGenericFinisherAnimation)
463 {
464 return EMeleeHitType.FINISHER_GENERIC;
465 }
466 else //specific hit depending on the component hit (gotten from the target)
467 {
468 return DetermineSpecificFinisherType(ItemBase.Cast(weapon));
469 }
470 }
471
472 return -1;
473 }

530 protected bool HitZoneSelectionRaycast(out vector hitPos, out int hitZone, out Object target, bool useCamera)
531 {
532 PlayerBase player = PlayerBase.Cast(m_DZPlayer);
533
534 vector pos;
535 vector dir;
536 vector playerDir;
537 if (useCamera)
538 { // What the player camera is looking at (crosshair)
539 vector cameraRotation;
540 player.GetCurrentCameraTransform(pos, dir, cameraRotation);
541 playerDir = dir;
542 }
543 else
544 { // What the player himself is looking at
545 playerDir = MiscGameplayFunctions.GetHeadingVector(player);
546 dir = GetGame().GetCurrentCameraDirection();
547 MiscGameplayFunctions.GetHeadBonePos(player, pos);
548 }
549
551 if (vector.Dot(dir, playerDir) < 0.5)
552 {
553 return false;
554 }
555
556 m_RayStart = pos;
557 m_RayEnd = m_RayStart + GetRange() * dir;
558
559 // raycast
560 set<Object> hitObjects = new set<Object>;
561 vector hitNormal;
562
563 if ( DayZPhysics.RaycastRV(m_RayStart, m_RayEnd, hitPos, hitNormal, hitZone, hitObjects, null, player, false, false, ObjIntersectIFire) && hitObjects.Count() > 0 )
564 {
565 target = hitObjects[0];
566 m_ForceUntargetable = false;
567
569 PlayerBase playerTarget = PlayerBase.Cast(target);
570 if (playerTarget && playerTarget.IsInVehicle())
571 {
572 if (vector.DistanceSq(pos, hitPos) > Math.SqrFloat(GetRange() * 0.5))
573 {
574 m_ForceUntargetable = true;
575 target = null;
576 hitPos = vector.Zero;
577 hitZone = -1;
578 }
579 }
580
581 return true;
582 }
583
584 return false;
585 }

589 {

591 PlayerBase player = PlayerBase.Cast(m_DZPlayer);
592
593 // ray properties
594 vector pos;
595 vector cameraDirection = GetGame().GetCurrentCameraDirection();
596
597 MiscGameplayFunctions.GetHeadBonePos(player, pos);
598 m_RayStart = pos;

655 {
656 // check direct visibility of object (obstruction check)

678 private bool IsEntityBehindEntityInAngle(EntityAI source, EntityAI target, float angle)
679 {
680 vector targetDirection = target.GetDirection();
681 ZombieBase targetZombie;
682 if (Class.CastTo(targetZombie, target))
683 {
684 targetDirection = Vector(targetZombie.GetOrientationSynced(),0,0);
685 targetDirection = targetDirection.AnglesToVector();
686 }
687 vector toSourceDirection = (source.GetPosition() - target.GetPosition());
688
689 targetDirection[1] = 0;
690 toSourceDirection[1] = 0;
691
692 targetDirection.Normalize();
693 toSourceDirection.Normalize();
694
695 float cosFi = vector.Dot(targetDirection, toSourceDirection);
696 vector cross = targetDirection * toSourceDirection;
697
698 int hitDir = Math.Acos(cosFi) * Math.RAD2DEG;
699
700 if (cross[1] < 0)
701 hitDir = -hitDir;
702
703 return hitDir <= (-180 + angle) || hitDir >= (180 - angle);
704 }
