722 void ToggleAnimation( string selection )
723 {
724 if ( m_State == PACKED )
725 {
726 return;
727 }
728
729 bool is_closed;

731
732 for ( int i = 0; i < m_ToggleAnimations.Count(); i++ )
733 {
734 ToggleAnimations toggle = m_ToggleAnimations.GetKey(i);
735
736 string toggle_off = toggle.GetToggleOff();
737 toggle_off.ToLower();
738 string toggle_on = toggle.GetToggleOn();
739 toggle_on.ToLower();
740
741 if ( toggle_off == selection || toggle_on == selection )
742 {
743 is_closed = m_OpeningMask & toggle.GetOpeningBit();
744 if ( is_closed )
745 {
746 SetAnimationPhase( toggle.GetToggleOff(), 0 );
747 AddProxyPhysics( toggle.GetToggleOff() );
748 SetAnimationPhase( toggle.GetToggleOn(), 1 );
749 RemoveProxyPhysics( toggle.GetToggleOn() );
750 m_ToggleAnimations.Set( toggle, false );
751 m_IsToggle = true;
752 m_OpeningMask &= ~toggle.GetOpeningBit();
753
754 if ( selection.Contains("window") )
755 {
756 ManipulateWindow();
757 }
758
759 if ( selection.Contains("entrance") || selection.Contains("door") )
760 {
761 ManipulateEntrance();
762 }
763
764 AnimateCamonetToggle(toggle);
765 }
766 else
767 {
768 SetAnimationPhase( toggle.GetToggleOff(), 1 );
769 RemoveProxyPhysics( toggle.GetToggleOff() );
770 SetAnimationPhase( toggle.GetToggleOn(), 0 );
771 AddProxyPhysics( toggle.GetToggleOn() );
772 m_ToggleAnimations.Set( toggle, true );
773 m_IsToggle = false;
774 m_OpeningMask |= toggle.GetOpeningBit();
775
776 if ( selection.Contains("window") )
777 {
778 ManipulateWindow();
779 }
780
781 if ( selection.Contains("entrance") || selection.Contains("door") )
782 {

786 AnimateCamonetToggle(toggle);
