415 protected Widget GetObjectByDegAngle( float deg_angle )
416 {
417 for ( int i = 0; i < m_RadialItemCards.Count(); ++i )
418 {
419 Widget w = m_RadialItemCards.GetKey( i );
420 float w_angle = GetAngleInDegrees( m_RadialItemCards.Get( w ) );
421 float offset = GetAngleInDegrees( m_AngleRadOffset ) / 2;
422 float min_angle = w_angle - offset;
423 float max_angle = w_angle + offset;
424
425 if ( min_angle < 0 ) min_angle += 360; //clamp 0-360
426 if ( max_angle > 360 ) max_angle -= 360;
427
428 if ( min_angle > max_angle ) //angle radius is in the cycling point 360->
429 {
430 if ( min_angle <= deg_angle ) //is cursor position also before this point
431 {
432 if ( deg_angle > max_angle )
433 {
434 return w;
435 }
436 }
437 else //is cursor position after this point
438 {
439 if ( deg_angle < max_angle )
440 {
441 return w;
442 }
443 }
444 }
445 else
446 {
447 if ( deg_angle >= min_angle && deg_angle < max_angle ) //min, max angles are within 0-360 radius
448 {
449 return w;
450 }
451 }
452 }

544 m_SelectedObject = NULL;

556 //show selector

559 }
560 //controller controls
561 else if ( IsUsingController() )
562 {
563 UpdataControllerInput();
564
565 //Controller tilt
566 if ( m_ControllerAngle > -1 && m_ControllerTilt > -1 )
567 {
568 //Right analogue stick
569 Widget w_selected = GetObjectByDegAngle( m_ControllerAngle );
570 //Select
571 if ( w_selected )

573 if ( w_selected != m_SelectedObject )
574 {
575 if ( m_ControllerTilt >= CONTROLLER_TILT_TRESHOLD_SELECT )
576 {
577 //Deselect

579

582 GetGame().GameScript.CallFunction( m_RegisteredClass, "OnControllerSelect", NULL, m_SelectedObject );
583 //show selector
584 ShowRadialSelector( m_SelectedObject );

586 }
587 }

591 m_SelectedObject = NULL;

594 }
595 }
