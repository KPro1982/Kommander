525 protected void CalculateBreathEffect(float timeSlice = 0, int type = -1, Param p = null)
526 {
527 float modifier = Math.Lerp(0.25, 0.5, m_BreathResidue);
528 float speed = timeSlice * modifier;
529 m_BreathResidue -= speed;
530 m_BreathResidue = Math.Clamp(m_BreathResidue,0,1);
531 float residue_final = Math.Lerp(0, 0.7, m_BreathResidue);
532
533 float hdr_mult;
534 hdr_mult = GetSceneHDRMul(0);
535 hdr_mult = Math.Clamp(hdr_mult,BREATH_HDR_MIN,BREATH_HDR_MAX);
536 hdr_mult = Math.InverseLerp(BREATH_HDR_MIN,BREATH_HDR_MAX,hdr_mult);
537 hdr_mult = Math.Lerp(BREATH_COLOR_MULT_MAX,BREATH_COLOR_MULT_MIN,hdr_mult);
538 m_BreathColor = ARGBF(0.0,1.0 * hdr_mult,1.0 * hdr_mult,1.0 * hdr_mult); //grayscaling of the image
539
540
541 m_BreathAlphaVal = Math.Lerp(m_BreathAlphaVal, residue_final, timeSlice);
542 }
