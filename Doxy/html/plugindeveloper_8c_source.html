329 void SpawnEntityOnGroundPatternGrid( PlayerBase player, string item_name, int count, float health, float quantity, int rows, int columns, float gapRow = 1, float gapColumn = 1)
330 {
331 if (!item_name)
332 {
333 return;
334 }
335 if ( GetGame().IsServer() )
336 {
337 float rowDist = 0;
338 float columnDist = 0;
339
340 vector playerPos = player.GetPosition();
341 vector camDirForward = player.GetDirection();
342 vector camDirRight = camDirForward.Perpend() * -1;
343 int countLoop = 0;
344 for (int i = 0; i < rows; i++)
345 {
346 vector posRow = playerPos + camDirForward * rowDist;
347 vector offsetSide;
348 columnDist = 0;
349 for (int j = 0; j < columns; j++)

354 SpawnEntityOnGroundPos(player, item_name, hlth, quantity, placement );
355 countLoop++;
356 if (countLoop == count)

359 }
360 columnDist += gapColumn;
361 }
362 rowDist += gapRow;
363 }

686 int QuickSortPartition( TStringArray arr, int left, int right )
687 {
688 string pivot = arr.Get( left );
689 int i = left;
690 int j = right + 1;
691 string temp;
692
693 while ( true )
694 {
695 while ( true )
696 {
697 i++;
698 if ( i > right || arr.Get(i) > pivot )
699 {
700 break;
701 }
702 }
703
704 while ( true )
705 {
706 j--;
707 if ( arr.Get(j) <= pivot )
708 {
709 break;
710 }
711 }
712
713 if ( i >= j )
714 {
715 break;
716 }
717
718 temp = arr.Get( i );
719 arr.Set( i, arr.Get(j) );
720 arr.Set( j, temp );
721 }
722
723 temp = arr.Get( left );
724 arr.Set( left, arr.Get(j) );
725 arr.Set( j, temp );
726
727 return j;
728 }
