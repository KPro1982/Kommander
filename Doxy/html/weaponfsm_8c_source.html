216 protected bool Internal_ValidateAndRepair()
217 {
218 bool repaired = false;
219
220 // Only repair stable states
221 WeaponStableState state = WeaponStableState.Cast(m_State);
222 if (state && state.IsRepairEnabled())
223 {
224 Weapon_Base weapon = state.m_weapon;
225 if (weapon)
226 {
227 repaired |= ValidateAndRepairHelper(weapon,
228 "MagazineRepair",
229 state.HasMagazine(), ( weapon.GetMagazine(0) != null ),
230 new WeaponEventAttachMagazine, new WeaponEventDetachMagazine,
231 state);
232
233 repaired |= ValidateAndRepairHelper(weapon,
234 "JammedRepair",
235 state.IsJammed(), weapon.IsJammed(),
236 new WeaponEventTriggerToJam, new WeaponEventUnjam,
237 state);
238
239 if (weapon.IsJammed())
240 return repaired;
241
242 int nMuzzles = weapon.GetMuzzleCount();
243 switch (nMuzzles)
244 {
245 case 1:
246 {
247 repaired |= ValidateAndRepairHelper(weapon,
248 "ChamberFiredRepair",
249 state.IsChamberFiredOut(0), weapon.IsChamberFiredOut(0),
250 new WeaponEventTrigger, new WeaponEventMechanism,
251 state);
252
253 repaired |= ValidateAndRepairHelper(weapon,
254 "ChamberRepair",
255 state.IsChamberFull(0), weapon.IsChamberFull(0),
256 new WeaponEventLoad1Bullet, new WeaponEventMechanism,
257 state);
258
259 break;
260 }
261 default:
262 {
263 for (int i = 0; i < nMuzzles; ++i)
264 {
265 repaired |= ValidateAndRepairHelper(weapon,
266 "ChamberFiredRepair",
267 state.IsChamberFiredOut(i), weapon.IsChamberFiredOut(i),
268 null, null, // A bit brute forced, not really any clean way to transition
269 state);
270
271 repaired |= ValidateAndRepairHelper(weapon,
272 "ChamberRepair",
273 state.IsChamberFull(i), weapon.IsChamberFull(i),
274 null, null, // A bit brute forced, not really any clean way to transition
275 state);
276 }
277
278 break;
279 }
280 }
281 }
282 }
283
284 return repaired;

287 protected bool ValidateAndRepairHelper(Weapon_Base weapon, string name, bool stateCondition, bool gunCondition, WeaponEventBase e1, WeaponEventBase e2, out WeaponStableState state)
288 {
289 if (LogManager.IsWeaponLogEnable()) { wpnDebugPrint("[wpnfsm] " + weapon.GetDebugName(weapon) + " ValidateAndRepair - " + name + " - " + m_State + " - state: " + stateCondition + " & weapon: " + gunCondition); }
290
291 if (stateCondition != gunCondition)
292 {
293 WeaponStableState repairedState;
294
295 // Seeing this message is not TOO bad, it just means this system is working
296 // It is simply being listed in the logs to identify how much the FSM state and weapon state still desyncs
297 // Which can be because of a myriad of causes, such as incorrectly set up transitions
298 // Or simply certain timings of certain actions or interrupts lined up perfectly, which can have repro rates such as 1/300
299 Error(string.Format("[wpnfsm] ValidateAndRepair Attempting to repair: %1 - %2 - %3 - state: %4 != weapon: %5",
300 weapon.GetDebugName(weapon), name, m_State, stateCondition, gunCondition));
301
302 if (e1 && e2)
303 repairedState = ValidateAndRepairStateFinder(gunCondition, e1, e2, state);
304
305 if (repairedState)
306 {
307 Terminate();
308 m_State = repairedState;
309 Start(null, true);
310 state = repairedState;
311 weapon.SyncSelectionState(state.HasBullet(), state.HasMagazine());
312 repairedState.SyncAnimState();
313
314 if (LogManager.IsWeaponLogEnable()) { wpnDebugPrint("[wpnfsm] " + weapon.GetDebugName(weapon) + " ValidateAndRepair - " + name + " - Result - " + m_State); }
315 return true;
316 }
317 else
318 {
319 // Last ditch effort
320 if (m_SynchronizeAttempts < MAX_SYNCHRONIZE_ATTEMPTS)
321 {
322 int currentTime = g_Game.GetTime();
323 int timeDiff = currentTime - m_LastSynchronizeTime;
324
325 // Careful with calling synchronize
326 if (timeDiff > MIN_SYNCHRONIZE_INTERVAL)
327 {
328 // If a lot of time passed since last attempt
329 // There is a possibility the weapon was fixed for a period
330 if (timeDiff > RESET_SYNCHRONIZE_THRESHOLD)
331 m_SynchronizeAttempts = 0;
332
333 // Only call this on server or in SP
334 // Synchronize will ask the server for its FSM state anyways
335 if (g_Game.IsServer())
336 weapon.RandomizeFSMState();
337
338 weapon.Synchronize();
339
340 ++m_SynchronizeAttempts;
341 m_LastSynchronizeTime = currentTime;
342 }
343 }
344 else
345 {
346 OnFailThresholdBreached(weapon, name, stateCondition, gunCondition);
347 }
348 }
349 }
350
351 return false;
352 }

473 void RandomizeFSMStateEx(array<MuzzleState> muzzleStates, bool hasMagazine, bool isJammed)
474 {
475 array<WeaponStableState> candidates = new array<WeaponStableState>;
476 int tc = m_Transitions.Count();
477 foreach (WeaponTransition trans : m_Transitions)
478 {
479 WeaponStableState state = WeaponStableState.Cast(trans.m_srcState);
480 if (state && state.HasMagazine() == hasMagazine && state.IsJammed() == isJammed)
481 {
482 if (state.IsSingleState())
483 {
484 // There is only one, insert it and stop
485 candidates.Insert(state);
486 break;
487 }
488
489 int nMuzzles = muzzleStates.Count();
490 int nMuzzlesState = state.GetMuzzleStateCount();
491 if (nMuzzles != nMuzzlesState)
492 {
493 ErrorEx(string.Format("Number of muzzles on the weapon (%1) does not correspond with what state has configured (%2).", nMuzzles, nMuzzlesState));
494 continue;
495 }
496
497 bool equal = true;
498 for (int i = 0; i < nMuzzles; ++i)
499 {
500 if (muzzleStates[i] != state.GetMuzzleState(i))
501 {
502 equal = false;
503 break;
504 }
505 }
506
507 if (equal)
508 candidates.Insert(state);
509 }
510 }
511
512 int cc = candidates.Count();
513 if (cc)
514 {
515 int randomIndex = Math.RandomInt(0, cc);
516 WeaponStableState selected = candidates.Get(randomIndex);
517 Terminate();
518 m_State = selected;
519 if (!Internal_ValidateAndRepair())
520 Start(null, true);
521 if (LogManager.IsWeaponLogEnable()) { wpnDebugPrint("[wpnfsm] RandomizeFSMState - randomized current state=" + m_State + " id=" + selected.GetCurrentStateID()); }
522 selected.SyncAnimState();
523 }
524 else
525 {
526 if (LogManager.IsWeaponLogEnable()) { wpnDebugPrint("[wpnfsm] RandomizeFSMState - warning - cannot randomize, no states available"); }
527 }
528 }
