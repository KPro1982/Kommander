439 // Handles subtle movement of the light point to create the effect of dancing shadows
440 void HandleDancingShadows(float time, float timeSlice)
441 {
442 if (m_DancingShadowsAmplitude > 0)
443 {
444 for (int i = 0; i < 3; i++ )
445 {
446 m_DancingShadowsLocalPos[i] = m_DancingShadowsLocalPos[i] + ( Math.RandomFloat(-m_DancingShadowsSpeed,m_DancingShadowsSpeed) * timeSlice) ;
447
448 if (m_DancingShadowsLocalPos[i] > m_DancingShadowsAmplitude)
449 m_DancingShadowsLocalPos[i] = m_DancingShadowsAmplitude;
450
451 if (m_DancingShadowsLocalPos[i] < -m_DancingShadowsAmplitude)
452 m_DancingShadowsLocalPos[i] = -m_DancingShadowsAmplitude;
453
454 }
455
456 if (m_Parent && !m_Parent.ToDelete())
457 {
458 // In order to move with the light, it must be detached from its parent first
459
460 m_Parent.RemoveChild(this);
461 SetPosition(m_LocalPos + m_DancingShadowsLocalPos);
462
463 m_Parent.AddChild(this, -1);
464 m_Parent.Update();
465 }
466
467 if (m_IsDebugEnabled)
468 {
469 Particle p = ParticleManager.GetInstance().PlayInWorld( ParticleList.DEBUG_DOT, GetPosition() );
470 p.SetParticleParam( EmitorParam.SIZE, 0.01);
471 }
472 }
473 else
474 {
475 m_DancingShadowsLocalPos = Vector(0,0,0);
476 }
477 }

635 // handles fading of brightness
636 private void HandleBrightnessFadeing(float timeSlice)
637 {
638 if ( m_Brightness != m_BrightnessTarget )
639 {
640 float brightness_difference = m_Brightness - m_BrightnessTarget;
641
642 if (brightness_difference > m_BrightnessSpeedOfChange*timeSlice)
643 brightness_difference = m_BrightnessSpeedOfChange*timeSlice;
644
645 if (brightness_difference < -m_BrightnessSpeedOfChange*timeSlice)
646 brightness_difference = -m_BrightnessSpeedOfChange*timeSlice;
647
648 m_Brightness -= brightness_difference;
649
650 if ( m_Brightness > 0 || m_BrightnessTarget > 0)
651 {
652 SetBrightness(m_Brightness * m_BrightnessPulse);
653 CorrectLightPulseDuringDaylight();
654 }
655 else
656 {
657 Destroy();
658 return;
659 }
660 }
661 else
662 {
663 SetBrightness(m_Brightness * m_BrightnessPulse);
664 CorrectLightPulseDuringDaylight();
665 }
666 }
