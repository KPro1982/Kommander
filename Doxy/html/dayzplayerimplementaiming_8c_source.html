285 protected float CalculateSwayMultiplier()
286 {
287 float max;
288 float time;
289 float time_clamped;
290 float ret;
291
292 if (m_PlayerPb.IsHoldingBreath())
293 {
294 time = m_TotalTime - m_ReferenceTime;
295
296 if (time < PlayerSwayConstants.SWAY_TIME_IN)
297 {
298 UpdateSwayState(eSwayStates.HOLDBREATH_IN);
299 max = PlayerSwayConstants.SWAY_TIME_IN;
300 time_clamped = Math.Clamp((m_TotalTime - m_SwayStateStartTime),0,max);
301 ret = Math.Lerp(m_LastSwayMultiplier,PlayerSwayConstants.SWAY_MULTIPLIER_STABLE,time_clamped/max);
302 }
303 else if (time >= PlayerSwayConstants.SWAY_TIME_IN && time < (PlayerSwayConstants.SWAY_TIME_IN + PlayerSwayConstants.SWAY_TIME_STABLE))
304 {
305 UpdateSwayState(eSwayStates.HOLDBREATH_STABLE);
306 ret = PlayerSwayConstants.SWAY_MULTIPLIER_STABLE;
307 }
308 else
309 {
310 UpdateSwayState(eSwayStates.HOLDBREATH_EXHAUSTED);
311 max = PlayerSwayConstants.SWAY_TIME_EXHAUSTED;
312 time_clamped = Math.Clamp((m_TotalTime - m_SwayStateStartTime),0,max);
313 ret = Math.Lerp(PlayerSwayConstants.SWAY_MULTIPLIER_STABLE,PlayerSwayConstants.SWAY_MULTIPLIER_EXHAUSTED,(time_clamped/max));
314 }
315 }
316 else
317 {
318 UpdateSwayState(eSwayStates.DEFAULT);
319 max = PlayerSwayConstants.SWAY_TIME_OUT;
320 time_clamped = Math.Clamp((m_TotalTime - m_SwayStateStartTime),0,max);
321 ret = Math.Lerp(m_LastSwayMultiplier,1,time_clamped/max);
322 }
323 return ret;
324 }
