122 void OnUpdateServer(float deltatime, float blood_scale, bool no_blood_loss )
123 {
124 m_ActiveTime += deltatime;
125
126 if(m_ActiveTime >= m_MaxTime)
127 {
128 if(m_Player.GetBleedingManagerServer() && !m_DeleteRequested)
129 {
130 m_Player.GetBleedingManagerServer().RequestDeletion(GetBit());//add yourself to a list of sources to be deleted
131 m_DeleteRequested = true;
132 }
133 }
134 if( !no_blood_loss )
135 {
136 float flow = m_FlowModifier;
137 switch( m_Type )
138 {
139 case eBleedingSourceType.NORMAL:
140 {
141 //do nothing
142 break;
143 }
144 case eBleedingSourceType.CONTAMINATED:
145 {
146 flow *= PlayerConstants.BLEEDING_SOURCE_BURN_MODIFIER;
147 }
148 }
149 m_Player.AddHealth("GlobalHealth","Blood", (PlayerConstants.BLEEDING_SOURCE_BLOODLOSS_PER_SEC * blood_scale * deltatime * flow) );
150 }
