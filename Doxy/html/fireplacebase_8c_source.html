1235 bool GetCeilingHeight( out float actual_height )
1236 {
1237 vector from = this.GetPosition();
1238 vector to = this.GetPosition();
1239 from[1] = from[1] + 1.0;
1240 to[1] = to[1] + MIN_CEILING_HEIGHT;
1241 vector contactPos;
1242 vector contactDir;
1243 int contactComponent;
1244
1245 bool hit = DayZPhysics.RaycastRV( from, to, contactPos, contactDir, contactComponent, NULL, NULL, this );
1246 actual_height = vector.Distance( from, contactPos ) + 1.0;
1247
1248 return hit;
1249 }
