449 SymptomBase QueueUpPrimarySymptom(int symptom_id, int uid = -1)
450 {
451 SymptomBase Symptom;
452 if((GetSymptomCount(symptom_id) >= GetSymptomMaxCount(symptom_id)) && GetSymptomMaxCount(symptom_id) != -1)
453 return null;
454 for( int i = 0; i < m_SymptomQueuePrimary.Count(); i++ )
455 {
456 if( m_SymptomQueuePrimary.Get(i).CanBeInterupted() && ComparePriority( GetSymptomPriority(symptom_id), m_SymptomQueuePrimary.Get(i).GetPriority() ) == 1 )
457 {
458 Symptom = SpawnSymptom( symptom_id, uid );
459 m_SymptomQueuePrimary.InsertAt(Symptom,i);
460
461 if( m_SymptomQueuePrimary.Count() > MAX_QUEUE_SIZE )
462 {
463 m_SymptomQueuePrimary.Get(MAX_QUEUE_SIZE).RequestDestroy();// no need to remove from the array, that's done via Symptom callback on destruct
464 }
465 break;
466 }
467 }
468 if( !Symptom && m_SymptomQueuePrimary.Count() < MAX_QUEUE_SIZE)
469 {
470 Symptom = SpawnSymptom( symptom_id, uid );
471 m_SymptomQueuePrimary.Insert(Symptom);
472 }
473 #ifdef DEVELOPER
474 SendServerDebugToClient();
475 #endif
476 return Symptom;
