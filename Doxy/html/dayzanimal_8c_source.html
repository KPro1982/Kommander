637 float ComputeHitDirectionAngleDeg(EntityAI source)
638 {
639 vector targetDirection = GetDirection();
640 vector toSourceDirection = (source.GetPosition() - GetPosition());
641
642 targetDirection[1] = 0;
643 toSourceDirection[1] = 0;
644
645 targetDirection.Normalize();
646 toSourceDirection.Normalize();
647
648 float cosFi = vector.Dot(targetDirection, toSourceDirection);
649 vector cross = targetDirection * toSourceDirection;
650
651 float dirAngleDeg = Math.Acos(cosFi) * Math.RAD2DEG;
652 if ( cross[1] < 0 )
653 dirAngleDeg = -dirAngleDeg;
654
655 return dirAngleDeg;
656 }

658 int TranslateHitAngleDegToDirectionIndex(float angleDeg)
659 {
660 if (angleDeg >= -20 && angleDeg <= 20) // front
661 {
662 return 1;
663 }
664 else if (angleDeg < 0) // left
665 {
666 return 2;
667 }
668
669 return 3; // right
670 }

672 int FindComponentDirectionOffset(string component)
673 {
674 const int directionCount = 4;
675
676 int offset = 0;

679 offset = 0;
