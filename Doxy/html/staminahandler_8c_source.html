392 void Update(float deltaT, int pCurrentCommandID)
393 {
394 #ifdef DEVELOPER
395 if (!m_StaminaEnabled || DiagMenu.GetBool(DiagMenuIDs.DM_CHEATS_STAMINA_DISABLE))
396 {
397 return;
398 }
399 #endif
400 if (m_Player)
401 {
402 // Calculates actual max stamina based on player's load
403 if (GetGame().IsServer() || !GetGame().IsMultiplayer())
404 {
406 m_PlayerLoad = m_Player.GetPlayerLoad();
407
409 if (m_PlayerLoad >= CfgGameplayHandler.GetStaminaWeightLimitThreshold())
410 {
411 m_StaminaCap = Math.Max((CfgGameplayHandler.GetStaminaMax() - (((m_PlayerLoad - CfgGameplayHandler.GetStaminaWeightLimitThreshold())/GameConstants.STAMINA_KG_TO_GRAMS) * CfgGameplayHandler.GetStaminaKgToStaminaPercentPenalty())),CfgGameplayHandler.GetStaminaMinCap());
412 }
413 else
414 {
415 m_StaminaCap = CfgGameplayHandler.GetStaminaMax();
416 }
417 }
418
419 // Calculates stamina gain/loss based on movement and load
420 m_Player.GetMovementState(m_State);
421
422 switch (pCurrentCommandID)
423 {
424 case DayZPlayerConstants.COMMANDID_MOVE:
425 StaminaProcessor_Move(m_State);
426 break;
427 case DayZPlayerConstants.COMMANDID_LADDER:
428 StaminaProcessor_Ladder(m_State);
429 break;
430 case DayZPlayerConstants.COMMANDID_SWIM:
431 StaminaProcessor_Swimming(m_State);
432 break;
433 case DayZPlayerConstants.COMMANDID_FALL:
434 case DayZPlayerConstants.COMMANDID_MELEE2:
435 case DayZPlayerConstants.COMMANDID_CLIMB:
436 break;
437 default:
438 if (!m_IsInCooldown)
439 {
440 m_StaminaDelta = GameConstants.STAMINA_GAIN_IDLE_PER_SEC;
441 }
442 break;
443 }
444
445 //Sets current stamina & stores + syncs data with client
446 float temp = m_StaminaDelta * deltaT;
447 if (temp < 0)
448 {
449 temp *= m_StaminaDepletionMultiplier;
450 }
451 else
452 {
453 temp *= m_StaminaRecoveryMultiplier;
454 }
455
456 m_Stamina = Math.Max(0, Math.Min((m_Stamina + temp), m_StaminaCap));
457 m_Stamina = m_Stamina - m_StaminaDepletion;
458
459 if (GetGame().IsServer() || !GetGame().IsMultiplayer())
460 {
461 m_Player.GetStatStamina().Set(m_Stamina);
462 m_Time += deltaT;
463
464 if (m_Time >= GameConstants.STAMINA_SYNC_RATE)
465 {
466 m_Time = 0;
467 SetStamina(m_Stamina);
468 }
469 }
470
471 #ifndef SERVER
472 m_Player.SetStamina(m_StaminaSynced, m_StaminaCap);
473 #endif
474
475 ApplyExhaustion();
476 CheckStaminaState();
477
478 m_StaminaDelta = 0;
479 m_StaminaDepletion = 0; // resets depletion modifier
480
481 }

876 void DepleteStamina(EStaminaModifiers modifier, float dT = -1)
877 {
878 #ifdef DEVELOPER
879 if (!m_StaminaEnabled || DiagMenu.GetBool(DiagMenuIDs.DM_CHEATS_STAMINA_DISABLE))
880 {
881 return;
882 }
883 #endif
884 float val = 0.0;
885 float current_time = m_Player.GetSimulationTimeStamp();
886 float time;
887 StaminaModifier sm = m_StaminaModifiers.GetModifierData(modifier);
888
889 // select by modifier type and drain stamina
890 switch (sm.GetType())
891 {
892 case m_StaminaModifiers.FIXED:
893 if (dT == -1)
894 {
895 dT = 1;
896 }
897 m_StaminaDepletion = m_StaminaDepletion + sm.GetMaxValue() * dT;
898 break;
899
900 case m_StaminaModifiers.RANDOMIZED:
901 val = Math.RandomFloat(sm.GetMinValue(), sm.GetMaxValue());
902 m_StaminaDepletion = m_StaminaDepletion + val;
903 break;
904
905 case m_StaminaModifiers.LINEAR:
906 if (!sm.IsInUse())
907 {
908 sm.SetStartTime(current_time + ( (PlayerSwayConstants.SWAY_TIME_IN + PlayerSwayConstants.SWAY_TIME_STABLE) / dT ) );
909 sm.SetRunTimeTick(dT);
910 sm.SetInUse(true);
911 }
912 time = Math.Clamp( ((current_time - sm.GetStartTime()) / sm.GetDurationAdjusted()), 0, 1 );
913 val = Math.Lerp(sm.GetMinValue(), sm.GetMaxValue(), time);
914 m_StaminaDepletion = m_StaminaDepletion + val;
915
916 break;
917
918 case m_StaminaModifiers.EXPONENTIAL:
919 if (!sm.IsInUse())
920 {
921 sm.SetStartTime(current_time + ( (PlayerSwayConstants.SWAY_TIME_IN + PlayerSwayConstants.SWAY_TIME_STABLE) / dT ) );
922 sm.SetRunTimeTick(dT);
923 sm.SetInUse(true);
924 }
925 time = Math.Clamp( ((current_time - sm.GetStartTime()) / sm.GetDurationAdjusted()), 0, 1 );
926 float exp;
927 if (sm.GetMinValue() < 1)
928 {
929 exp = 1 - Math.Lerp(0, sm.GetMaxValue(), time);
930 }
931 else
932 {
933 exp = Math.Lerp(0, sm.GetMaxValue(), time);
934 }
935 val = Math.Pow(sm.GetMinValue(),exp);
936 m_StaminaDepletion = m_StaminaDepletion + val;
937
938 break;
939 }
940
942 SetCooldown(sm.GetCooldown(),modifier);
943 m_StaminaDepletion = Math.Clamp(m_StaminaDepletion, 0, CfgGameplayHandler.GetStaminaMax());
944
945 m_StaminaDepletion = m_StaminaDepletion * m_StaminaDepletionMultiplier;
946 }
