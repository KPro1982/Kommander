551 {
552 delta *= DayZGame.Cast(GetGame()).GetFoodDecayModifier();
553 m_DecayDelta += ( 1 + ( 1 - GetHealth01( "", "" ) ) );
554 if ( hasRootAsPlayer )

568 switch ( GetFoodStageType() )
569 {

571 m_DecayTimer = ( GameConstants.DECAY_FOOD_RAW_FRVG + ( Math.RandomFloat01() * ( GameConstants.DECAY_FOOD_RAW_FRVG * ( GameConstants.DECAY_TIMER_RANDOM_PERCENTAGE / 100.0 ) ) ) );
572 m_LastDecayStage = FoodStageType.RAW;
