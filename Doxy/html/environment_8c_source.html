319 protected float GetEnvironmentTemperature()
320 {
321 float temperature;
322 temperature = g_Game.GetMission().GetWorldData().GetBaseEnvTemperature();
323 temperature += Math.AbsFloat(temperature * m_Clouds * GameConstants.ENVIRO_CLOUDS_TEMP_EFFECT);
324
325 if (IsWaterContact())
326 {
327 temperature -= Math.AbsFloat(temperature * GameConstants.ENVIRO_WATER_TEMPERATURE_COEF);
328 }
329
330 if (IsInsideBuilding())
331 {
332 temperature += Math.AbsFloat(temperature * GameConstants.ENVIRO_TEMPERATURE_INSIDE_COEF);
333 }
334 else if (IsInsideVehicle())
335 {
336 temperature += Math.AbsFloat(temperature * GameConstants.ENVIRO_TEMPERATURE_INSIDE_VEHICLE_COEF);
337 }
338 else if (IsUnderRoof())
339 {
340 temperature += Math.AbsFloat(temperature * GameConstants.ENVIRO_TEMPERATURE_UNDERROOF_COEF);
341 temperature -= GameConstants.ENVIRO_TEMPERATURE_WIND_COEF * GetWindModifierPerSurface() * m_Wind;
342 }
343 else
344 {
345 temperature -= GameConstants.ENVIRO_TEMPERATURE_WIND_COEF * GetWindModifierPerSurface() * m_Wind;
346 temperature -= Math.AbsFloat(temperature * m_Fog * GameConstants.ENVIRO_FOG_TEMP_EFFECT);
347 temperature -= GetTemperatureHeightCorrection();
348 }
349
350 // incorporate temperature from temperature sources (buffer)
351 if (Math.AbsFloat(m_UTSAverageTemperature) > 0.001)

359 // Calculats wet/drying delta based on player's location and weather
360 protected float GetWetDelta()
361 {
362 float wetDelta = 0;
363 if ( IsWaterContact() )
364 {
366 if (m_WaterLevel >= WATER_LEVEL_HIGH)
367 {
368 wetDelta = 1;
369 }
370 else if (m_WaterLevel >= WATER_LEVEL_MID && m_WaterLevel < WATER_LEVEL_HIGH)
371 {
372 wetDelta = 0.66;
373 }
374 else if (m_WaterLevel >= WATER_LEVEL_LOW && m_WaterLevel < WATER_LEVEL_MID)
375 {
376 wetDelta = 0.66;
377 }
378 else if (m_WaterLevel >= WATER_LEVEL_NONE && m_WaterLevel < WATER_LEVEL_LOW)
379 {
380 wetDelta = 0.33;
381 }
382 }
383 else if (IsRaining() && !IsInsideBuilding() && !IsUnderRoof() && !IsInsideVehicle())
384 {
386 wetDelta = GameConstants.ENVIRO_WET_INCREMENT * GameConstants.ENVIRO_TICKS_TO_WETNESS_CALCULATION * (m_Rain) * (1 + (GameConstants.ENVIRO_WIND_EFFECT * m_Wind));
387 }
388 else
389 {
391 float tempEffect = Math.Max(m_PlayerHeat + GetEnvironmentTemperature(), 1.0);
392 if (m_HeatSourceTemp != 0)
393 {
394 tempEffect *= GameConstants.ENVIRO_FIRE_INCREMENT;
395 }
396
397 float weatherEffect = ((1 - (m_Fog * GameConstants.ENVIRO_FOG_DRY_EFFECT))) * (1 - (m_Clouds * GameConstants.ENVIRO_CLOUD_DRY_EFFECT));
398 if (weatherEffect <= 0)
399 {
400 weatherEffect = 1.0;
401 }
402
403 wetDelta = -(GameConstants.ENVIRO_DRY_INCREMENT * weatherEffect * tempEffect);
404 if (!IsInsideBuilding())
405 {
406 wetDelta *= 1 + (GameConstants.ENVIRO_WIND_EFFECT * m_Wind);
407 }
408 }
409
410 return wetDelta;
411 }

703 protected void ProcessItemsHeat()
704 {
705 float hcHead, hcBody, hcFeet;
706 float hHead, hBody, hFeet;
707
708 float heatComfortAvg;
709 float heatAvg;
710
711 BodyPartHeatProperties(m_HeadParts, GameConstants.ENVIRO_HEATCOMFORT_HEADPARTS_WEIGHT, hcHead, hHead);
712 BodyPartHeatProperties(m_BodyParts, GameConstants.ENVIRO_HEATCOMFORT_BODYPARTS_WEIGHT, hcBody, hBody);
713 BodyPartHeatProperties(m_FeetParts, GameConstants.ENVIRO_HEATCOMFORT_FEETPARTS_WEIGHT, hcFeet, hFeet);
714
715 heatComfortAvg = (hcHead + hcBody + hcFeet) / 3;
716 heatAvg = (hHead + hBody + hFeet) / 3;
717 heatAvg = heatAvg * GameConstants.ENVIRO_ITEM_HEAT_TRANSFER_COEF;
718
719 // heat buffer
720 float applicableHB = 0.0;
721 if (m_UTSAverageTemperature < 0.001)
722 {
723 applicableHB = m_Player.GetStatHeatBuffer().Get() / 30.0;
724 if (applicableHB > 0.0)
725 {
726 if (m_HeatBufferTimer > 1.0)
727 {
728 m_Player.GetStatHeatBuffer().Add(Math.Min(EnvTempToCoef(m_EnvironmentTemperature), -0.1) * GameConstants.ENVIRO_PLAYER_HEATBUFFER_DECREASE);
729 }
730 else
731 {
732 m_HeatBufferTimer += GameConstants.ENVIRO_PLAYER_HEATBUFFER_TICK;
733 }
734 }
735 else
736 {
737 m_HeatBufferTimer = 0.0;
738 }
739 }
740 else
741 {
742 applicableHB = m_Player.GetStatHeatBuffer().Get() / 30.0;
743 if (m_HeatComfort > PlayerConstants.THRESHOLD_HEAT_COMFORT_MINUS_WARNING)
744 {
745 m_Player.GetStatHeatBuffer().Add(GameConstants.ENVIRO_PLAYER_HEATBUFFER_INCREASE);
746 m_HeatBufferTimer = 0.0;
747 }
748 else
749 {
750 m_HeatBufferTimer = 0.0;
751 }
752 }
753
754 m_HeatComfort = (heatComfortAvg + heatAvg + (GetPlayerHeat() / 100)) + EnvTempToCoef(m_EnvironmentTemperature);
755 if ((m_HeatComfort + applicableHB) < (PlayerConstants.THRESHOLD_HEAT_COMFORT_PLUS_WARNING - 0.01))
756 {
757 m_HeatComfort += applicableHB;
758 }
759 else
760 {
761 if (m_HeatComfort <= (PlayerConstants.THRESHOLD_HEAT_COMFORT_PLUS_WARNING - 0.01))
762 {
763 m_HeatComfort = PlayerConstants.THRESHOLD_HEAT_COMFORT_PLUS_WARNING - 0.01;
764 }
765 }
766
767 m_HeatComfort = Math.Clamp(m_HeatComfort, m_Player.GetStatHeatComfort().GetMin(), m_Player.GetStatHeatComfort().GetMax());
768
769 m_Player.GetStatHeatComfort().Set(m_HeatComfort);
770 }

803 attachment = m_Player.GetInventory().GetAttachmentFromIndex(attIdx);
804 if (attachment.IsClothing())
805 {
806 item = ItemBase.Cast(attachment);
807 int attachmentSlot = attachment.GetInventory().GetSlotId(0);
808
810 for (int i = 0; i < pBodyPartIds.Count(); i++)
811 {
812 if (attachmentSlot == pBodyPartIds.Get(i))
813 {
814 float heatIsoMult = 1.0;
815 if (attachmentSlot == InventorySlots.VEST)
816 {
817 heatIsoMult = GameConstants.ENVIRO_HEATISOLATION_VEST_WEIGHT;
818 }
819
820 if (attachmentSlot == InventorySlots.BACK)
821 {
822 heatIsoMult = GameConstants.ENVIRO_HEATISOLATION_BACK_WEIGHT;
823 }
824
825 pHeatComfort += heatIsoMult * MiscGameplayFunctions.GetCurrentItemHeatIsolation(item);
826
827 // go through any attachments and cargo (only current level, ignore nested containers - they isolate)
828 int inAttCount = item.GetInventory().AttachmentCount();
829 if (inAttCount > 0)
830 {
831 for (int inAttIdx = 0; inAttIdx < inAttCount; inAttIdx++)
832 {
833 EntityAI inAttachment = item.GetInventory().GetAttachmentFromIndex(inAttIdx);
834 ItemBase itemAtt = ItemBase.Cast(inAttachment);
835 if (itemAtt)
836 {
837 pHeat += itemAtt.GetTemperature();
838 }
839 }
840 }
841 if (item.GetInventory().GetCargo())
842 {
843 int inItemCount = item.GetInventory().GetCargo().GetItemCount();
844
845 for (int j = 0; j < inItemCount; j++)
846 {
847 ItemBase inItem;
848 if (Class.CastTo(inItem, item.GetInventory().GetCargo().GetItem(j)))
849 {
850 pHeat += inItem.GetTemperature();
851 }
852 }
853 }
854 }
855 }
856 }
857 }
858
859 pHeatComfort = (pHeatComfort / pBodyPartIds.Count()) * pCoef;
860 pHeat = (pHeat / pBodyPartIds.Count()) * pCoef;
861 }

893 protected void ProcessTemperatureSources()
894 {
895 if (m_UTemperatureSources.Count() == 0)
896 {
897 m_UTSAverageTemperature = m_UTSAverageTemperatureBuffer.Add(0);
898
899 return;
900 }
901
902 array<float> utsTemperatures = new array<float>();
903
904 // get temperature from the source (based on distance), save it for min/max filtering
905 for (int i = 0; i < m_UTemperatureSources.Count(); i++)
906 {
907 utsTemperatures.Insert(CalcTemperatureFromTemperatureSource(m_UTemperatureSources[i]));
908 }
909
910 float min = MiscGameplayFunctions.GetMinValue(utsTemperatures);
911 float max = MiscGameplayFunctions.GetMaxValue(utsTemperatures);
912
913 if (max > 0 && min < 0)
914 {
916 m_UTSAverageTemperature = m_UTSAverageTemperatureBuffer.Add((max + min) * 0.5);
917 }
918 else
919 {
920 m_UTSAverageTemperature = m_UTSAverageTemperatureBuffer.Add(max);
921 }

929 float CalcTemperatureFromTemperatureSource(notnull UTemperatureSource uts)
930 {
931 float distance = vector.Distance(m_Player.GetPosition(), uts.GetPosition());
932 distance = Math.Max(distance, 0.1); //min distance cannot be 0 (division by zero)
933 float temperature = 0;
934
935 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::distance: %1", distance), "Environment");
936
938 if (distance > uts.GetFullRange())
939 {
940 float distFactor = 1 - (distance / uts.GetMaxRange());
941 distFactor = Math.Max(distFactor, 0.0);
942 temperature = uts.GetTemperature() * distFactor;
943 //temperature = Math.Clamp(temperature, uts.GetTemperatureMin(), uts.GetTemperatureMax());
944 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::distFactor: %1", distFactor), "Environment");
945 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::temperatureXX: %1", temperature), "Environment");
946 }
947 else
948 {
949 temperature = uts.GetTemperature();
950 }
951
952 //Debug.Log(string.Format("CalcTemperatureFromTemperatureSource::temperature: %1", temperature), "Environment");
953
954 return temperature;
