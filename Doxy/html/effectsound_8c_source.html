387 protected bool SoundWaveValidation()
388 {
389 bool valid = true;
390
391 if ( m_SoundFadeInDuration > GetSoundWaveLength() )
392 {
393 SoundError("FadeIn is longer than sound wave length.");
394 valid = false;
395 }
396
397 if ( m_SoundFadeOutDuration > GetSoundWaveLength() )
398 {
399 SoundError("FadeOut is longer than sound wave length.");
400 valid = false;

404 {
405 SoundError("FadeIn & FadeOut are longer than sound wave length.");
406 valid = false;

444 override void Event_OnFrameUpdate(float time_delta)
445 {
446 if ( IsSoundPlaying() )
447 {
448 // Make the sound follow the parent
449 if ( m_SoundParent )
450 {
451 SetCurrentPosition( m_SoundParent.ModelToWorld( GetLocalPosition() ) );
452 }
453
454 // FadeIn
455 if ( m_SoundWaveStarting )
456 {
457 if ( m_SoundFadeInDuration > 0 )
458 {
459 SetSoundVolume( GetSoundVolume() + (time_delta / m_SoundFadeInDuration) );
460
461 if ( GetSoundVolume() >= m_SoundWaveVolumeMax )
462 {
463 Event_OnSoundFadeInStopped();
464 SetSoundVolume( m_SoundWaveVolumeMax );
465 m_SoundWaveStarting = false;
466 }
467 }
468 else
469 {
470 SetSoundVolume( m_SoundWaveVolumeMax );
471 m_SoundWaveStarting = false;
472 }
473 }
474
475 // FadeOut
476 if ( m_SoundWaveStopping )
477 {
478 if ( m_SoundFadeOutDuration > 0 )
479 {
480 if ( m_SoundFadeOutInitVolume == 0 )
481 {
482 m_SoundFadeOutInitVolume = GetSoundVolume();
483 Event_OnSoundFadeOutStarted();
484 }
485 SetSoundVolume( GetSoundVolume() - (m_SoundFadeOutInitVolume / m_SoundFadeOutDuration * time_delta) );
486 }
487 else
488 {
489 SetSoundVolume( 0 );
490 }
491
492 if ( GetSoundVolume() <= 0 )
493 {
494 if ( m_SoundWaveObject )
495 {
496 m_SoundWaveObject.Stop();
497 }
498 }
499 }
500
501 // Counting timer here because loop play
502 m_SoundWaveTime += time_delta;
503 }
