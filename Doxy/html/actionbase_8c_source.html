685 bool Can( PlayerBase player, ActionTarget target, ItemBase item, int condition_mask )
686 {
687 if ( ( (condition_mask & m_ConditionMask) != condition_mask ) || ( !IsFullBody(player) && !player.IsPlayerInStance(GetStanceMask(player)) ) || player.IsRolling() )
688 return false;
689
690 if ( HasTarget() )
691 {
692 if(!FirearmActionBase.Cast(this))
693 {
694 EntityAI entity = EntityAI.Cast(target.GetObject());
695 if ( entity && !target.GetObject().IsMan() )
696 {
697 Man man = entity.GetHierarchyRootPlayer();
698 if ( man && man != player )
699 return false;
700 }
701 }
702
703 if ( m_ConditionTarget && !m_ConditionTarget.Can(player, target))
704 return false;
705 }
706
707 if ( m_ConditionItem && !m_ConditionItem.Can(player, item))
708 return false;
709
710 if ( !ActionCondition(player, target, item) )
711 return false;
712
713 if ( IsFullBody(player) )
714 {
715 int stanceIdx = DayZPlayerUtils.ConvertStanceMaskToStanceIdx(GetStanceMask(player));
716 if (stanceIdx != -1 && !DayZPlayerUtils.PlayerCanChangeStance(player, stanceIdx ))
717 return false;
718 }
719
720 return true;
721 }

784 bool InventoryReservation(ActionData action_data)
785 {
786 if( (IsLocal() || !UseAcknowledgment()) && IsInstant() )
787 return true;

789 //action_data.m_ReservedInventoryLocations = new array<ref InventoryLocation>;

792 InventoryLocation handInventoryLocation = NULL;
793
794 // lock target if it has target
795 if( HasTarget() )
796 {
797 ItemBase targetItem;
798 if ( ItemBase.CastTo(targetItem,action_data.m_Target.GetObject()) )
799 {

808 action_data.m_Player.GetInventory().AddInventoryReservationEx( targetItem, targetInventoryLocation, GameInventory.c_InventoryReservationTimeoutMS);
809 }
810 }
811 }

814 handInventoryLocation.SetHands(action_data.m_Player,action_data.m_Player.GetItemInHands());

820 else

827 }

831 if( targetInventoryLocation )
832 action_data.m_ReservedInventoryLocations.Insert(targetInventoryLocation);
