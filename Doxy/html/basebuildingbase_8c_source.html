1040 void CalcDamageAreaRotation( float angle_deg, out vector center, out vector orientation )
1041 {
1042 if ( angle_deg != 0 )
1043 {
1044 //orientation
1045 orientation[0] = orientation[0] - angle_deg;
1046
1047 //center
1048 vector rotate_axis;
1049 if ( MemoryPointExists( "rotate_axis" ) )
1050 {
1051 rotate_axis = ModelToWorld( GetMemoryPointPos( "rotate_axis" ) );
1052 }
1053 float r_center_x = ( Math.Cos( angle_deg * Math.DEG2RAD ) * ( center[0] - rotate_axis[0] ) ) - ( Math.Sin( angle_deg * Math.DEG2RAD ) * ( center[2] - rotate_axis[2] ) ) + rotate_axis[0];
1054 float r_center_z = ( Math.Sin( angle_deg * Math.DEG2RAD ) * ( center[0] - rotate_axis[0] ) ) + ( Math.Cos( angle_deg * Math.DEG2RAD ) * ( center[2] - rotate_axis[2] ) ) + rotate_axis[2];
1055 center[0] = r_center_x;
1056 center[2] = r_center_z;
1057 }
