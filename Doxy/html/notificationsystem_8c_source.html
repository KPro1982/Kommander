253 types.Insert( type_curr );
254 type_curr++;

256
257 for ( int i = 0; i < m_Instance.m_DataArray.Count(); i++ )
258 {
259 types.RemoveItem( m_Instance.m_DataArray.GetKey( i ) );

261

266 NotificationData data2 = new NotificationData( "please_add_an_icon", "please_add_a_title", "optional_description" );
