135 //Print(excludeMask);

259 {

261 float increase = accoDiff * m_AccoInterpolationSpeed * timeSlice;
262 m_EyeAcco += increase;

372 private void OnUndergroundPresenceUpdate(EUndergroundPresence newPresence, EUndergroundPresence oldPresence)
373 {
374 //Print("-----> On Undeground Presence update " + EnumTools.EnumToString(EUndergroundPresence, newPresence) + " " + EnumTools.EnumToString(EUndergroundPresence, oldPresence));
375 if (newPresence > EUndergroundPresence.NONE)
376 {
377 if (oldPresence == EUndergroundPresence.NONE)
378 {
379 EnableLights(true);
380 }
381 if (newPresence > EUndergroundPresence.OUTER && oldPresence <= EUndergroundPresence.OUTER)
382 {
383 GetGame().GetWeather().SuppressLightningSimulation(true);
384 m_Player.PlaySoundSetLoop(m_AmbientSound, "Underground_SoundSet",3,3);
385 }
386 if (newPresence == EUndergroundPresence.FULL)
387 {
388 m_AnimTimerLightBlend = new AnimationTimer();
389 m_AnimTimerLightBlend.Run(1, this, "OnUpdateTimerIn", "OnUpdateTimerEnd",0, false, LIGHT_BLEND_SPEED_IN);
390 }
391 }
392 if (newPresence < EUndergroundPresence.FULL && oldPresence == EUndergroundPresence.FULL)
393 {
394 m_AnimTimerLightBlend = new AnimationTimer();
395 m_AnimTimerLightBlend.Run(0, this, "OnUpdateTimerOut", "OnUpdateTimerEnd",m_LightingLerp, false, LIGHT_BLEND_SPEED_OUT);
396 }
397 if (newPresence <= EUndergroundPresence.OUTER && oldPresence > EUndergroundPresence.OUTER)
398 {
399 GetGame().GetWeather().SuppressLightningSimulation(false);
400 if (m_AmbientSound)
401 m_Player.StopSoundSet(m_AmbientSound);
402 }
403 if (newPresence == EUndergroundPresence.NONE && oldPresence >= EUndergroundPresence.OUTER)
404 {
405 GetGame().GetWorld().SetUserLightingLerp(0);
406 EnableLights(false);
407 }
