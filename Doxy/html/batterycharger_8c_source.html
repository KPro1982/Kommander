50 override void OnWork( float consumed_energy )
51 {
52 // Charging functionality
53 ItemBase battery = ItemBase.Cast( GetCompEM().GetPluggedDevice() );
54
55 if ( battery )
56 {
57 if ( GetGame().IsServer() )
58 {
59 float battery_capacity = battery.GetCompEM().GetEnergyMax();
60
61 if ( battery.GetCompEM().GetEnergy() < battery_capacity )
62 {
63 // Heat up the items so players know they are working.
64 this.SetTemperature(60);
65 battery.SetTemperature(60);
66
67 float charger_health = GetHealth("", "");
68 float energy_add = m_ChargeEnergyPerSecond * ( consumed_energy / GetCompEM().GetEnergyUsage() );
69
70 if ( GetCompEM().ConsumeEnergy(energy_add) ) // consumes energy from the power source
71 {
72 // There is enough of energy to use
73 energy_add = energy_add * ( 0.5 + charger_health*0.005 ); // Damaged charger works less efficiently - 50% damage causes 75% efficiency
74 }
75 else
76 {
77 // There is NOT enough of energy to use
78 energy_add = 0;
79 }
80
81 battery.GetCompEM().AddEnergy( energy_add );
82 }
83 else
84 {
85 battery.GetCompEM().SetEnergy( battery_capacity );
86 }
87 m_BatteryEnergy0To100 = battery.GetCompEM().GetEnergy0To100();
88 SetSynchDirty();
89 }
90 }

111 void UpdateStatusLights()
112 {
113 if ( GetGame().IsClient() || !GetGame().IsMultiplayer() )
114 {
115 if (GetCompEM().IsWorking())
116 {
117 SwitchLightOn();
118 ItemBase battery = ItemBase.Cast( GetCompEM().GetPluggedDevice() );
119
120 if (battery)
121 {
122 RedLightOff();
123
124 if (m_BatteryEnergy0To100 <= 33)
125 {
126 // Less than 1/3 charged, yellow status light must repeatedly blink
127
128 if (m_BlinkingStatusLightIsOn)
129 YellowLightOn();
130 else
131 YellowLightOff();
132
133 m_BlinkingStatusLightIsOn = !m_BlinkingStatusLightIsOn;
134 }
135 else if (m_BatteryEnergy0To100 > 33 && m_BatteryEnergy0To100 <= 66)
136 {
137 // Less than 2/3 charged, yellow status light must glow
138
139 YellowLightOn();
140 }
141 else if (m_BatteryEnergy0To100 > 66 && m_BatteryEnergy0To100 < 100)
142 {
143 // Less than 3/3 charged, yellow status light must glow, green light must blink
144
145 YellowLightOn();
146
147 if (m_BlinkingStatusLightIsOn)
148 GreenLightOn();
149 else
150 GreenLightOff();
151
152 m_BlinkingStatusLightIsOn = !m_BlinkingStatusLightIsOn;
153 }
154 else if (m_BatteryEnergy0To100 >= 100)
155 {
156 // Fully charged, green light must glow
157 YellowLightOff();
158 GreenLightOn();
159 }
160 }
161 else
162 {
163 if (m_BlinkingStatusLightIsOn)
164 RedLightOn();
165 else
166 RedLightOff();
167
168 m_BlinkingStatusLightIsOn = !m_BlinkingStatusLightIsOn;
169
170 GreenLightOff();
171 YellowLightOff();
172 }
173 }
174 else
175 {
176 SwitchLightOff();
177 GreenLightOff();
178 RedLightOff();
179 YellowLightOff();
180 }
181 }
