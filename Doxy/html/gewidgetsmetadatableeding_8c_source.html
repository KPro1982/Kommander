255 vector GenerateSequenceRandomPosition()
256 {
257 if (m_PositioningFramesArray)
258 {
259 int arrayCount = m_PositioningFramesArray.Count();
260 int index = Math.RandomInt(0,arrayCount);
261 if (index == m_LastPositionFrameUsed)
262 {
263 index++;
264 if (index >= arrayCount)
265 {
266 index = 0;
267 }
268 }
269
270 Widget frame = m_PositioningFramesArray[index];
271 m_LastPositionFrameUsed = index;
272
273 if (frame)
274 {
275 int screenX,screenY;
276 float x,y,x_size,y_size;
277 frame.GetScreenPos(x,y);
278 frame.GetScreenSize(x_size,y_size);
279
280 x = Math.RandomFloatInclusive(x,x+x_size);
281 y = Math.RandomFloatInclusive(y,y+y_size);
282
283 return Vector(x,y,0);
284 }
285 }
286 return vector.Zero;
