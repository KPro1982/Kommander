169 protected float ComputeAttackUtility(DayZInfectedAttackType pAttackType, float pTargetDistance, int pPitch, float pProbability)
170 {
171 // pitch
172 if ( pAttackType.m_Pitch != pPitch )
173 return 0;
174
175 // distance
176 float distDiff = ( pAttackType.m_Distance ) - pTargetDistance;
177 if ( distDiff < 0 )
178 return 0;
179 float distDiffFrac = distDiff / 10;
180 float utilityDistance = (1 - distDiffFrac) * 100; // distance is most important
181
182 // probability
183 float probDiff = pAttackType.m_Probability - pProbability;
184 if ( probDiff < 0 )
185 return 0;
186 float utilityProbability = (1 - probDiff) * 10; // distance is most important
187
188 // total
189 float util = utilityDistance + utilityProbability;
190 //Print("Attack Utility " + util);
191
192 return util;
193 }
