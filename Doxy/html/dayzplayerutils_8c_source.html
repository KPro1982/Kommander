135 if (sideDist > tgAngle) // out of cone

279 static bool HandleStoreCartridge(DayZPlayer player, Weapon_Base weapon, int muzzleIndex, float damage, string cartTypeName, string magTypeName, bool CanDrop = true)
280 {
281 if (damage < 1.0)
282 {
283 // find suitable heap / mag
284 ref array<Magazine> mags = new array<Magazine>;
285 if (DayZPlayerUtils.FindMagazinesForAmmo(player, magTypeName, mags))

287 int health_lvl = -1;
288 float test_heatlh = 1 - damage;
289 int sz = mags.Count();
290 for (int i = 0; i < sz; ++i)

292 Magazine mag_i = mags.Get(i);
293 // add small amout for
294
295 if( health_lvl == -1)
296 {
297 if (mag_i.CanAddCartridges(1) /*&& Math.AbsFloat(1 - mag_i.GetHealthLevelValue(mag_i.GetHealthLevel()) - damage) < 0.01*/)

299 int num_health_lvl = mag_i.GetNumberOfHealthLevels();
300 for( int j = 1; j < num_health_lvl; j++)
301 {
302 if (mag_i.GetHealthLevelValue(j) < test_heatlh )
303 {
304 health_lvl = j - 1;
305 break;
306 }
307 }
308 }
309 }
310
311 if( mag_i.GetHealthLevel() == health_lvl )
312 {
313 if (mag_i.ServerStoreCartridge(damage, cartTypeName))
314 return true;
315 }
316 }
317 }
318
319 // create a new one in inventory
320 InventoryLocation inv_loc = new InventoryLocation;
321 if (player.GetInventory().FindFirstFreeLocationForNewEntity(magTypeName, FindInventoryLocationType.ANY, inv_loc))
322 {
323 EntityAI eai_inv = GetGame().SpawnEntity(magTypeName, inv_loc, ECE_IN_INVENTORY, RF_DEFAULT);
324 if (eai_inv && eai_inv.IsInherited(Magazine))
325 {
326 Magazine mag_inv;
327 if (Class.CastTo(mag_inv, eai_inv))
328 {
329 mag_inv.ServerSetAmmoCount(0);
330 mag_inv.SetHealth("", "", (1 - damage) * mag_inv.GetMaxHealth());

414 static EWaterLevels CheckWaterLevel(DayZPlayer pPlayer, out vector waterLevel)
415 {
416 SHumanCommandSwimSettings swimData = pPlayer.GetDayZPlayerType().CommandSwimSettingsW();
417 vector pp = pPlayer.GetPosition();
418 waterLevel = HumanCommandSwim.WaterLevelCheck(pPlayer, pp);
419
420 if (waterLevel[1] < swimData.m_fToCrouchLevel)
421 {
422 return EWaterLevels.LEVEL_LOW;

427 }
428 else// if (waterLevel[1] >= swimData.m_fToErectLevel)
