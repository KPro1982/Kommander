269 {
270 if ( !OnlineServices.ErrorCaught( error ) )
271 {
272 OnlineServices.SetBiosUser( joiner );
273
274 SelectUserEx( joiner );
275 OnlineServices.SetSessionHandle( handle );
276
277 if ( g_Game.GetGameState() == DayZGameState.IN_GAME )
278 {
279 g_Game.SetLoadState( DayZLoadState.JOIN_START );
280 OnlineServices.GetSession();
281 }
282 else
283 {
284 if ( GetGame().GetUIManager() && GetGame().GetInput().IsActiveGamepadSelected() )
285 {
286 GetGame().GetUIManager().CloseMenu( MENU_TITLE_SCREEN );
287 GetGame().GetInput().IdentifyGamepad( GamepadButton.BUTTON_NONE );
288 }
289 g_Game.SetLoadState( DayZLoadState.JOIN_START );
290 g_Game.GamepadCheck();
291 }
292 }
293 else
294 {
295 if ( g_Game.GetGameState() != DayZGameState.IN_GAME )
296 {
297 if ( GetGame().GetMission() )
298 {
299 if ( g_Game.GetGameState() != DayZGameState.MAIN_MENU )
300 {
301 GetGame().GetUIManager().CloseAllSubmenus();
302 GetGame().GetMission().AbortMission();
303 if (g_Game.GetGameState() == DayZGameState.JOIN)
304 NotificationSystem.AddNotification(NotificationType.JOIN_FAIL_GET_SESSION, 6);
305 g_Game.SetGameState( DayZGameState.MAIN_MENU );
306 g_Game.SetLoadState( DayZLoadState.MAIN_MENU_START );
307 g_Game.GamepadCheck();
308 return;
309 }
310 }
311 else
312 {
313 g_Game.MainMenuLaunch();
314 }
315 }
