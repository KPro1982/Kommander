203 void OnGamepadIdentification(int gamepad)
204 {
205 if( gamepad > -1 )

207 DayZLoadState state = g_Game.GetLoadState();
208
209 g_Game.DeleteGamepadDisconnectMenu();
210 SelectActiveGamepad( gamepad );
211 g_Game.SelectUser( gamepad );
212 g_Game.SetPreviousGamepad( gamepad );
213 if( state == DayZLoadState.MAIN_MENU_START || state == DayZLoadState.MAIN_MENU_USER_SELECT )

215 if( GetGame().GetMission() )
216 GetGame().GetMission().Reset();
